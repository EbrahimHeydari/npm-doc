npm -l                                                // نمایش لیست تمام دستورات اصلی به همراه توضیحات

npm init                                              // مقدار دهی اولیه 

npm init - y | --yes                                  // مقدار دهی اولیه بودن پرسش 

npm -v | --version                                    // نمایش ورژن

npm config ls -l                                      // نمایش لیست تمام مقادیر ممکن کانفیگ

npm set init-author-name "author-name"                // تغییر نام پیشفرض کد نویس

npm set init-license "license-name"                   // تغییر لایسنس پیشفرض

npm i | install package-name                          // نصب پکیج

npm i | install package-name -D                       // نصب پکیج در حالت توسعه

npm i | install package-name@1.0.0                    // نصب نسخه ای خاص از یک ورژن

npm rm | remove | un | uninstall package-name         // حذف پکیج

npm rm | remove | un | uninstall package-name -D      // حذف در حالت توسعه

npm root                                              // نمایش آدرس نصب پکیج ها

npm list                                              // نمایش لیست پکیج ها به همراه وابستگی ها

npm list --depth 0                                    // نمایش لیست پکیج ها بدون وابستگی ها

npm audit fix                                         // رفع مشکلات امنیتی پکیج ها
